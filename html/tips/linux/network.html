ネットワーク設定

グラフィカルに設定

ターミナル 
sudo nmtui

コマンドで設定

ターミナル 
nmcli device                                                        # デバイス名(ネットワークインターフェース名)を確認
sudo nmcli connection show                                          # 接続プロファイル一覧を確認

# 新しい接続プロファイルを作成
sudo nmcli connection add type [接続タイプ ethernet wifi 等]    \
    con-name [接続プロファイル名]                               \
    ifname [デバイス名]

# 接続プロファイルを編集
# ipv4.method manual            : IPv4手動設定
# ipv4.ignore-auto-dns yes      : DHCPサーバーから自動的に提供されるDNSサーバー情報を無視する
# ipv6.method ignore            : IPv6 無効化

sudo nmcli connection modify [接続プロファイル名]               \
    ipv4.method manual                                        \
    ipv4.addresses [IPアドレス]/[ビットマスク]                  \
    ipv4.gateway [デフォルトゲートウェイ]                       \
    ipv4.dns [DNSサーバのIPアドレス]                           \
    ipv4.ignore-auto-dns yes                                  \
    ipv6.method ignore

sudo nmcli connection reload
sudo nmcli connection up [接続プロファイル名]

NAT設定

IPテーブル設定に必要なパッケージをインストール

ターミナル 
sudo apt install iptables-persistent

IPテーブルの状態を確認

ターミナル 
sudo iptables -Z    # パケットカウンタリセット
sudo iptables -L -v # IPテーブルの状態を確認

IPテーブルのルールを削除

ターミナル 
sudo iptables -F FORWARD            # 既存のFORWARDチェーンのルールをすべて削除
sudo iptables -t nat -F             # 既存のNATテーブルのルールをすべて削除

NATを許可する

ターミナル 
sudo nano /etc/sysctl.conf

エディタ 
net.ipv4.ip_forward=1

ターミナル 
sudo sysctl -p

IPテーブル設定 (iptables)

ターミナル 

# ルール設定

# 外部から出る全てのパケットの送信元IPをルータの[外部インターフェース]のIPに変換
sudo iptables                                   \
    -t nat                                      \
    -A POSTROUTING                              \
    -o [外部インターフェース]                     \
    -j MASQUERADE

# 外部から内部への応答パケットを許可
sudo iptables                                   \
    -A FORWARD                                  \
    -i [外部インターフェース]                     \
    -o [内部インターフェース]                     \
    -m state --state RELATED,ESTABLISHED        \
    -j ACCEPT

# 内部から外部へのパケットをすべて許可
sudo iptables                                   \
    -A FORWARD                                  \
    -i [内部インターフェース]                     \
    -o [外部インターフェース]                     \
    -j ACCEPT

# ルールの恒久化
sudo netfilter-persistent save

MACアドレスの固定化

ターミナル 
sudo nano /etc/udev/rules.d/71-pen-net.rules

エディタ 
SUBSYSTEM=="net", ACTION=="add", ATTR{address}=="[固定したいMACアドレス]", NAME="[インターフェース名]"

ターミナル 
sudo udevadm control --reload
sudo udevadm trigger

IPテーブル設定 (iptables IPアドレス指定)

ターミナル 
# ルール設定
# 内部 → 外部 NAT
sudo iptables                                   \
    -t nat                                      \
    -A POSTROUTING                              \
    -s [内部ネットワークアドレス/プレフィックス]    \
    -j MASQUERADE

# 外部 → 内部：戻り通信のみ許可
sudo iptables                                   \
    -A FORWARD                                  \
    -d [内部ネットワークアドレス/プレフィックス]    \
    -m state --state RELATED,ESTABLISHED        \
    -j ACCEPT

# 内部 → 外部：すべて許可
sudo iptables                                   \
    -A FORWARD                                  \
    -s [内部ネットワークアドレス/プレフィックス]    \
    -j ACCEPT

# ルールの恒久化
sudo netfilter-persistent save

IPテーブル設定 (nftables)

ターミナル 
sudo apt install nftables
sudo nano /etc/nftables.conf

エディタ 
#!/usr/sbin/nft -f

flush ruleset

define LAN_IP   = [内部ネットワークアドレス/プレフィックス]

# table テーブルタイプ テーブル名 {}
#       テーブルタイプ
#           ip    : IPv4
#           ip6   : IPv6
#           inet  : IPv4 + IPv6
#       テーブル名 : 任意

table inet filter {

    # ルータ自身への通信
    chain input {
        type filter hook input priority filter;

        # 戻り通信 : 許可
        ct state established,related accept

        # ループバック : 許可
        iifname "lo" accept

        # ICMP (IPv4)
        ip protocol icmp accept

        # ICMPv6 (IPv6)
        ip6 nexthdr icmpv6 accept

        # TCP 許可 {SSH, HTTP}
        tcp dport {22, 80} accept

        # UDP 許可 {NTP, DHCP}
        udp dport {123, 67} accept
        
        # 必要なポートを調査した後有効化
        # drop
    }
    
    # ルータ自身から外への通信
    chain output {
        type filter hook output priority filter;
    }

    chain forward {
        # type filter hook forward priority filter;
        type filter hook forward priority 0;

        # 内部 → 外部：すべて許可
        ip saddr $LAN_IP accept

        # 外部 → 内部：戻り通信のみ許可
        ct state established,related accept

        # それ以外はすべて破棄
        drop
    }
}

table ip nat {
    chain postrouting {
        type nat hook postrouting priority 100;

        # 内部 → 外部 : MASQUERADE
        ip saddr $LAN_IP masquerade
    }
}

ターミナル 
sudo systemctl enable nftables
sudo systemctl restart nftables
sudo nft list ruleset               # 現在のルールを一覧表示

ファイアウォール設定

インストール

ターミナル 
sudo apt install ufw
sudo ufw enable

状態確認

ターミナル 
sudo ufw status
sudo ufw status numbered    # ID付きで確認

ルールを消す

ターミナル 
sudo ufw delete [ID]

ポートを開ける

ターミナル 
sudo ufw allow [ポート番号]                     # [ポート番号]のポートを開ける
sudo ufw allow [ポート番号]/udp                 # [ポート番号]のUDPポートを開ける
sudo ufw allow [ポート番号]/tcp                 # [ポート番号]のTCPポートを開ける
sudo ufw allow [ポート番号A]:[ポート番号B]/tcp  # [ポート番号A]～[ポート番号B]のTCPポートを開ける
