Arch Linux のセットアップ方法

キー配列変更

ターミナル 
ls /usr/share/kbd/keymaps/**/*.map.gz | grep jp  # 日本語キーボードに対応しているか確認
loadkeys jp106

パッケージ更新

ターミナル 
curl -o /etc/pacman.d/mirrorlist "https://archlinux.org/mirrorlist/?country=all&protocol=https&use_mirror_status=on"        # 使用可能なミラーサイトを取得
sed -i '/## Japan/,/##/ s/^#Server/Server/' /etc/pacman.d/mirrorlist                                                        # 日本のミラーサイトを有効化
sed -i '/## Worldwide/,/##/ s/^#Server/Server/' /etc/pacman.d/mirrorlist                                                    # World Wideのミラーサイトを有効化

# 証明書キーの更新
pacman-key --init
pacman-key --populate archlinux
pacman-key --refresh-keys

# rm -rf /var/cache/pacman/pkg/*                                                                                            # パッケージが壊れた場合

pacman -Syy                                                                                                                 # パッケージデータベースの更新

パーティションの作成とマウント設定

ターミナル 
fdisk -l                                                                                                                    # 認識されているストレージの確認

############ レガシーBIOS ##################
fdisk /dev/[ルートにしたいストレージ]
g                                                                                                                           # GPTパーティションテーブルの作成
n                                                                                                                           # 新しいパーティションの作成
p                                                                                                                           # プライマリパーティションの作成
Enter                                                                                                                       # パーティション番号の指定(デフォルトでOK)
Enter                                                                                                                       # 開始セクタの指定(デフォルトでOK)
+1M                                                                                                                         # 終了セクタの指定(1MB)
Y                                                                                                                           # signatureの上書き
t                                                                                                                           # パーティションのタイプ変更
EF02                                                                                                                        # BIOS Boot Partitionに変更
w                                                                                                                           # パーティションテーブルの保存と終了
############################################

############ UEFI ##########################
fdisk /dev/[ルートにしたいストレージ]
g                                                                                                                           # GPTパーティションテーブルの作成
n                                                                                                                           # 新しいパーティションの作成
p                                                                                                                           # プライマリパーティションの作成
Enter                                                                                                                       # パーティション番号の指定(デフォルトでOK)
Enter                                                                                                                       # 開始セクタの指定(デフォルトでOK)
+512M                                                                                                                       # 終了セクタの指定(512MB)
Y                                                                                                                           # signatureの上書き
t                                                                                                                           # パーティションのタイプ変更
EF00                                                                                                                        # EFI System Partitionに変更
w                                                                                                                           # パーティションテーブルの保存と終了
############################################


fdisk /dev/[ルートにしたいストレージ]
n                                                                                                                           # 新しいパーティションの作成
p                                                                                                                           # プライマリパーティションの作成
Enter                                                                                                                       # パーティション番号の指定(デフォルトでOK)
Enter                                                                                                                       # 開始セクタの指定(デフォルトでOK)
Enter                                                                                                                       # 終了セクタの指定(デフォルトでOK)
w                                                                                                                           # パーティションテーブルの保存と終了
mkfs.ext4 /dev/[ルートにしたいパーティション]                                                                                  # ext4でフォーマット

mount /dev/[ルートにしたいパーティション] /mnt


############ UEFI ##########################
mount --mkdir /dev/[ルートにしたいストレージの最初のパーティション] /mnt/boot
############################################

pacstrap /mnt base linux linux-firmware                                                                                     # ベースシステムのインストール

echo KEYMAP=jp106 > /mnt/etc/vconsole.conf                                                                                  # コンソールのキー配列を日本語キーボードに設定
genfstab -U /mnt >> /mnt/etc/fstab                                                                                          # fstabの生成
arch-chroot /mnt                                                                                                            # 新しいシステムにchroot

初期設定

ターミナル 
timedatectl set-timezone Asia/Tokyo                                                                                         # タイムゾーンの設定

pacman -Syu
pacman -S grub efibootmgr sudo nano
mkinitcpio -P                                                                                                               # initramfsの生成


############ レガシーBIOS ##################
grub-install --target=i386-pc /dev/[ルートにしたいストレージ]
grub-mkconfig -o /boot/grub/grub.cfg
############################################

############ UEFI ##########################
grub-install --target=x86_64-efi --efi-directory=/boot --bootloader-id=GRUB --removable
grub-mkconfig -o /boot/grub/grub.cfg
############################################


# ホスト名設定
hostnamectl set-hostname [新しいホスト名]
echo 127.0.1.1       [新しいホスト名] >> /etc/hosts                                                                         # 127.0.1.1が確実に無い場合
# sed -i '/^127\.0\.1\.1/ s/.*/127.0.1.1    [新しいホスト名]/; t; $ a 127.0.1.1    [新しいホスト名]' /etc/hosts               # 安全版

echo LANG=ja_JP.UTF-8 > /etc/locale.conf

passwd

useradd -m pen
passwd pen

sudo mkdir -p /etc/profile.d
echo 'export EDITOR=nano' | sudo tee /etc/profile.d/editor.sh

権限変更

ターミナル 
visudo

エディタ (コメントアウトを外す) 
%wheel ALL=(ALL:ALL) ALL

ターミナル 
gpasswd -a pen wheel

必要なパッケージをインストール

ターミナル 
pacman -S networkmanager noto-fonts noto-fonts-cjk noto-fonts-emoji man git
pacman -S plasma kde-applications sddm chromium fcitx5 fcitx5-im fcitx5-mozc                                                # (全てEnter)

systemctl enable sddm
systemctl enable NetworkManager
systemctl restart NetworkManager
reboot

NetworkManagerをインストールし忘れて再起動してしまった場合

ターミナル 
sudo nano /etc/systemd/network/20-wired.network

エディタ 
[Match]
Name=eth0

[Network]
DHCP=yes

ターミナル 
sudo systemctl enable --now systemd-networkd
sudo systemctl enable --now systemd-resolved
pacman -S networkmanager

sudo systemctl stop systemd-networkd
sudo systemctl disable systemd-networkd

sudo nmtui
systemctl enable NetworkManager
systemctl restart NetworkManager

日本語入力

ターミナル 
sed -i 's/#ja_JP.UTF-8 UTF-8/ja_JP.UTF-8 UTF-8/' /etc/locale.gen
locale-gen

mkdir -p ~/.config/autostart
cp /usr/share/applications/org.fcitx.Fcitx5.desktop ~/.config/autostart/
sudo nano /etc/environment

エディタ 
GTK_IM_MODULE=fcitx
QT_IM_MODULE=fcitx
XMODIFIERS=@im=fcitx
SDL_IM_MODULE=fcitx

スワップサイズの変更

ターミナル 
sudo mkswap -U clear --size 4G --file /swapfile
sudo swapon /swapfile
sudo nano /etc/fstab
/swapfile none swap defaults 0 0
