TigerVNCサーバの構築(ディスプレイが無い場合)

TigerVNCのインストール

ターミナル 
sudo apt install -y tigervnc-standalone-server tigervnc-common tigervnc-tools openbox obconf xfce4 xfce4-goodies

パスワード設定

ターミナル 
vncpasswd

設定の変更

ターミナル 
mkdir -p ~/.config/tigervnc
nano ~/.config/tigervnc/xstartup

エディタ 
#!/bin/sh
export XMODIFIERS="@im=fcitx"
export GTK_IM_MODULE=fcitx
export QT_IM_MODULE=fcitx
export SDL_IM_MODULE=fcitx
export DISPLAY=:1

xsetroot -solid grey
xhost +SI:localuser:$(whoami)

exec startxfce4

起動設定

ターミナル 
sudo nano /etc/systemd/system/tigervnc@:1.service

エディタ 
[Unit]
Description=TigerVNC Server for %i
After=network.target

[Service]
Type=forking
User=[ユーザ名]
Group=[ユーザ名]
WorkingDirectory=/home/[ユーザ名]

ExecStart=/usr/bin/vncserver %i -localhost no
ExecStop=/usr/bin/vncserver -kill %i

Restart=always

[Install]
WantedBy=multi-user.target
